					tagName = nodeList[i].tagName.toLowerCase();
					if (_elementsKeys.indexOf(tagName) !== -1) {
						if (typeof(_elements[tagName][cbName]) === "function") {
							try {
								// console.log({tagName, cbName});
								_elements[tagName][cbName].apply(nodeList[i], []);
							} catch (er) {

							}
